#!/bin/bash
FILE=$1; RES=$2; OUT=$3

hex=`convert $FILE -colors 16 -format "%c" histogram:info:|\
       sort -nr | head -1 | tr -s ' ' | cut -d' ' -f7`

cp $FILE $OUT/$FILE

convert -extent $RES -resize $RES -gravity SouthEast \
        -background "$hex" $OUT/$FILE $new 

echo "Image created, saved to $OUT/FILE"
